128static float CalcularAnguloDesviacion(float ax, float ay, float az)
129{
130 //Producto punto entre vector actual y vector de referencia
131 float dot = (ax * base_x) + (ay * base_y) + (az * base_z);
132 //Magnitud de ambos vectores
133 float mag_base = sqrtf(base_x * base_x + base_y * base_y + base_z * base_z);
134 float mag_actual = sqrtf(ax * ax + ay * ay + az * az);
135 //Coseno del ángulo entre ambos vectores
136 float cos_theta = dot / (mag_base * mag_actual);
137
138 // Evitar valores fuera de rango [-1,1] por errores de redondeo
139 if (cos_theta > 1.0f)
140 cos_theta = 1.0f;
141 if (cos_theta < -1.0f)
142 cos_theta = -1.0f;
143 //Convertir de rad a grados
144 return acosf(cos_theta) * (180.0f / 3.14159f);
145}

170void LeerAcelerometro(void *pvParameter)
171{
172 float suma_x = 0, suma_y = 0, suma_z = 0;
173 uint16_t muestras = 0;
174
175 TickType_t start_time = xTaskGetTickCount();
176
177 while (true)
178 { //Leer valores del acelerometro
179 float ax = ReadXValue();
180 float ay = ReadYValue();
181 float az = ReadZValue();
182 // Aplicar filtro de suavizado
183 datos_acelerometro.ax = FiltroSuavizado(ax, datos_acelerometro.ax);
184 datos_acelerometro.ay = FiltroSuavizado(ay, datos_acelerometro.ay);
185 datos_acelerometro.az = FiltroSuavizado(az, datos_acelerometro.az);
186
187 // Calibración inicial
188 if (!calibrado)
189 {
190 suma_x += datos_acelerometro.ax;
191 suma_y += datos_acelerometro.ay;
192 suma_z += datos_acelerometro.az;
193 muestras++;
194 //Verifica si terminó el tiempo de calibración
195 if ((xTaskGetTickCount() - start_time) >= pdMS_TO_TICKS(TIEMPO_CALIBRACION))
196 { //Calcula los promerios como valores de referencia
197 base_x = suma_x / muestras;
198 base_y = suma_y / muestras;
199 base_z = suma_z / muestras;
200 calibrado = true;
201 printf("✅ Calibracion completa: X=%.2f Y=%.2f Z=%.2f\r\n", base_x, base_y, base_z);
202 }
203 }
204 else
205 { // Calcular ángulo de desviación respecto a la posición de referencia.
206 datos_acelerometro.angulo = CalcularAnguloDesviacion(
207 datos_acelerometro.ax,
208 datos_acelerometro.ay,
209 datos_acelerometro.az);
210 }
211
212 // Espera hasta el siguiente periodo de muestreo.
213 vTaskDelay(1000 / PERIODO_MUESTREO_AC);
214 }
215}

226void ProcesarPostura(void *pvParameter)
227{
228
229 while (true)
230 {
231 if (calibrado)
232 {
233 // Verificar si el ángulo supera el umbral de inclinación
234 if (fabs(datos_acelerometro.angulo) > UMBRAL_INCLINACION)
235 { //Acumula tiempo en mala postura
236 bad_posture_time += PERIODO_MUESTREO_AC;
237
238 //Actualiza estado según el tiempo acumulado
239 if (bad_posture_time >= TIEMPO_ALERTA)
240 posture_state = 2;
241 else if (bad_posture_time >= TIEMPO_ADVERTENCIA)
242 posture_state = 1;
243 }
244 else
245 {
246 //Postura correcta: reiniciar contador y estado
247 bad_posture_time = 0;
248 posture_state = 0;
249 }
250 }
251 // Esperar al siguiente periodo
252 vTaskDelay(pdMS_TO_TICKS(PERIODO_MUESTREO_AC));
253 }
